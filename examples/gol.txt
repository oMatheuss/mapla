extern func malloc(size: int): void*;
extern func free(mem: int*);
extern func memcpy(dst: int*, src: int*, cnt: int): int*;
extern func putchar(ch: int): int;
extern func printf(fmt: char*, ...);
extern func usleep(ms: int);
extern func srand(seed: int);
extern func rand(): int;

var WIDTH: int  = 32;
var HEIGHT: int = 16;

var DEAD_CELL: int = 32;
var ALIVE_CELL: int = 35;

func printBoard(board: int*) do
    for y to HEIGHT do
        var dy = y * WIDTH;
        for x to WIDTH do
            putchar(board[dy + x]);
        end
        putchar(10);
    end
end

func createBoard(): int* do
    var board = malloc(WIDTH * HEIGHT * 4) as int*;
    for y to HEIGHT do
        var dy = y * WIDTH;
        for x to WIDTH do
            board[dy + x] = DEAD_CELL;
        end
    end
    return board;
end

func wrapAround(value: int, maxValue: int): int do
    if value >= maxValue then value -= maxValue; end
    else if value < 0 then value += maxValue; end
    return value;
end

func checkNeighbors(board: int*, posX: int, posY: int): int do
    var sum = 0;
    for j = -1 to 2 do
        var y = wrapAround(posY + j, HEIGHT);
        var dy = y * WIDTH;
        for i = -1 to 2 do
            if i != 0 || j != 0 then
                var x = wrapAround(posX + i, WIDTH);
                if board[dy + x] == ALIVE_CELL then
                    sum += 1;
                end
            end
        end
    end
    return sum;
end

func updateBoard(actualState: int*) do
    var nextState = createBoard();
    memcpy(nextState, actualState, WIDTH * HEIGHT * 4);
    for y to HEIGHT do
        var dy = y * WIDTH;
        for x to WIDTH do
            var actual = actualState[dy + x];
            var alives = checkNeighbors(actualState, x, y);
            if actual == ALIVE_CELL && alives < 2 || alives > 3 then
                nextState[dy + x] = DEAD_CELL;
            end
            if actual == DEAD_CELL && alives == 3 then
                nextState[dy + x] = ALIVE_CELL;
            end
        end
    end
    memcpy(actualState, nextState, WIDTH * HEIGHT * 4);
    free(nextState);
end

func randomizeBoard(board: int*) do
    srand(1283);
    for y to HEIGHT do
        var dy = y * WIDTH;
        for x to WIDTH do
            if rand() > 15000 then
                board[dy + x] = ALIVE_CELL;
            end
        end
    end
end

func clearTerminal() do
    printf("%c[H", 27);
    printf("%c[J", 27);
end

func main(): int do
    var board = createBoard();
    randomizeBoard(board);

    while true do
        clearTerminal();
        printBoard(board);
        usleep(100000);

        updateBoard(board);
    end

    return 0;
end
