struct Color r: byte, g: byte, b: byte, a: byte end

extern func InitWindow(width: int, height: int, title: char*);
extern func SetTargetFPS(fps: int);
extern func WindowShouldClose(): bool;
extern func BeginDrawing();
extern func ClearBackground(color: Color);
extern func DrawText(text: char*, posX: int, posY: int, fontSize: int, color: Color);
extern func EndDrawing();

func main(): int do
    var BLACK = { r: 0 as byte, g: 0 as byte, b: 0 as byte, a: 0xFF as byte };
    var RED   = { r: 0xFF as byte, g: 0 as byte, b: 0 as byte, a: 0xFF as byte };

    InitWindow(800, 450, "RayLib Example");
    SetTargetFPS(60);

    while !WindowShouldClose() do
        BeginDrawing();
        ClearBackground(BLACK);
        DrawText("Hello World", 190, 200, 20, RED);
        EndDrawing();
    end

    return 0;
end
