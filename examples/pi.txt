extern func printf(fmt: char*, arg: int);

func main(): int do
    real sum = 0.0;
    real flip = -1.0;
    real i = 1.0;

    for j to 100000 do
        flip = -flip;
        sum += flip / ( 2.0 * i - 1.0 );
        i += 1.0;
    end
    
    printf("PI = %d", (sum * 4.0 * 100.0) as int);
    return 0;
end
